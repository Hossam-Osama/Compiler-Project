METHOD_BODY = STATEMENT_LIST
STATEMENT_LIST = STATEMENT STATEMENT_LIST_PRIME
STATEMENT = DECLARATION | IF | WHILE | ASSIGNMENT
DECLARATION = PRIMITIVE_TYPE 'id' ';'
PRIMITIVE_TYPE = 'int' | 'float'
IF = 'if' '(' EXPRESSION ')' '{' STATEMENT '}' 'else' '{' STATEMENT '}'
WHILE = 'while' '(' EXPRESSION ')' '{' STATEMENT '}'
ASSIGNMENT = 'id' '=' EXPRESSION ';'
EXPRESSION = SIMPLE_EXPRESSION EXPRESSION_PRIME1
SIMPLE_EXPRESSION = TERM SIMPLE_EXPRESSION_PRIME | SIGN TERM SIMPLE_EXPRESSION_PRIME
TERM = FACTOR TERM_PRIME
FACTOR = 'id' | 'num' | '(' EXPRESSION ')'
SIGN = '+' | '-'
STATEMENT_LIST_PRIME = STATEMENT STATEMENT_LIST_PRIME | '\L'
SIMPLE_EXPRESSION_PRIME = 'addop' TERM SIMPLE_EXPRESSION_PRIME | '\L'
TERM_PRIME = 'mulop' FACTOR TERM_PRIME | '\L'
EXPRESSION_PRIME1 = '\L' | 'relop' SIMPLE_EXPRESSION
